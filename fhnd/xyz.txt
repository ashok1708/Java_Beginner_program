IMPORT JAVA.IO.*;
IMPORT JAVA.UTIL.*;

CLASS FILEDEMO
{
 PUBLIC STATIC VOID MAIN(STRING...ARG) THROWS IOEXCEPTION
 {
  FILE F =NEW FILE(ARG[0]);

  IF(!F.EXISTS())
  {
   SYSTEM.ERR.PRINTLN("\NFILE NOT FOUND - "+ARG[0]+"\N");
   SYSTEM.EXIT(2);
  }
 
  SYSTEM.OUT.PRINTLN();

  SYSTEM.OUT.PRINTLN("FILENAME                  : "+F.GETNAME());
  SYSTEM.OUT.PRINTLN("PARENT NAME               : "+F.GETPARENT());
  SYSTEM.OUT.PRINTLN("PATH                      : "+F.GETPATH());
  SYSTEM.OUT.PRINTLN("ABSOLUTE PATH             : "+F.GETABSOLUTEPATH());
  SYSTEM.OUT.PRINTLN("ABSOLUTE PATH             : "+F.GETCANONICALPATH());
  SYSTEM.OUT.PRINTLN("PATH IS ABSOLUTE          : "+F.ISABSOLUTE());
  SYSTEM.OUT.PRINTLN("FILE IS A REGULAR FILE?); : "+F.ISFILE());
  SYSTEM.OUT.PRINTLN("FILE IS A DIRECTORY?      : "+F.ISDIRECTORY());
  SYSTEM.OUT.PRINTLN("FILE IS HIDDEN ?          : "+F.ISHIDDEN());
  SYSTEM.OUT.PRINTLN("FILE IS READABLE ?        : "+F.CANREAD());
  SYSTEM.OUT.PRINTLN("FILE IS WRITABLE ?        : "+F.CANWRITE());
  SYSTEM.OUT.PRINTLN("FILE IS EXECUTABLE ?      : "+F.CANEXECUTE());
  SYSTEM.OUT.PRINTLN("FILE SIZE                 : "+F.LENGTH());
  SYSTEM.OUT.PRINTLN("DATE OF LAST MODIFICATION : "+NEW DATE(F.LASTMODIFIED()));




  SYSTEM.OUT.PRINTLN();
 }
}
